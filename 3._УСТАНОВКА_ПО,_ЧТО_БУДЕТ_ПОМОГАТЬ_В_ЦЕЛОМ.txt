СПИСОК ПО И УСТАНОВКА:

1. "sudo pacman -Syu" - ОБНОВЛЕНИЕ ВСЕХ ПАКЕТОВ, ОШИБКА КОТОРАЯ МОЖЕТ ВОЗНИКНУТЬ:
	1) "sudo pacman-key --edit-key "key" save" - В СЛУЧАЕ КОГДА НЕ ХОЧЕТ ОБНОВЛЯТЬ ИЗ-ЗА НЕПРАВИЛЬНОЙ ПОДПИСИ, ВВЕСТИ KEY(НАЙТИ В ОШИБКЕ "sudo pacman -Syu");
	2) "sudo pacman-key --init && sudo pacman-key --populate archlinux && sudo pacman-key --refresh-keys" - ЗАПУСК ПРОВЕРКИ И ОБНОВЛЕНИЯ КЛЮЧЕЙ GPG(ПРОВЕРЯЮТ ПОДПИСИ У ПАКЕТОВ).

2. YAY - ПОЗВОЛЯЕТ УСТАНАВЛИВАТЬ ПАКЕТЫ ИЗ AUR
	1) "sudo pacman -S git base-devel" - ИНСТРУМЕНТЫ ДЛЯ СКАЧИВАНИЯ И РАБОТЫ YAY;
	2) КЛОНИРУЕМ YAY ИЗ РЕПОЗИТОРИЯ AUR С ПОМОЩЬЮ ПРЯМОЙ 	ССЫЛКИ: "git clone https://aur.archlinux.org/yay.git";
	3) "cd yay" ЗАЙДЕМ В ПАПКУ;
	4) "makepkg -si" РАСПАКУЕТ И УСТАНОВИТ YAY;
	5) "~/.cache/yay:$" - РАСПОЛОЖЕНИЕ СОБРАННЫХ ПАКЕТОВ;
	6) КОМАНДЫ ДЛЯ YAY:
	"yay -Syu" ОБНОВИТ ВСЕ ПАКЕТЫ В СИСТЕМЕ;
	"yay -Yc" ОЧИСТИТ НЕЖЕЛАТЕЛЬНЫЕ ЗАВИСИМОСТИ;
	"yay -Ps" ВЫВОДИТ СИСТЕМНУЮ СТАТИСТИКУ.

3. УСТАНОВКА ДРАЙВЕРОВ:
	1) "lspci -v | grep -A1 -e VGA -e 3D" ОПРЕДЕЛИТ ВИДЕОКАРТУ
	2) "sudo pacman -S mesa" - ОСНОВА;
	3) "sudo pacman -S lib32-mesa" - ПОДДЕРЖКА 32-БИТНЫХ ПРИЛОЖЕНИЙ;
	4) "sudo pacman -S xf86-video-amdgpu" - ОБЕСПЕЧИВАЕТ 2D УСКОРЕНИЕ, ОСНОВНОЙ ДРАЙВЕР.

4. УСТАНОВКА I3-WM(ОКОННЫЙ МЕНЕДЖЕР):
	1) "sudo pacman -S xorg-server xorg-drivers" - УСТАНАВЛИВАЕМ ИКСЫ И СВОБОДНЫЕ ДРАЙВЕРА;
	2) "sudo pacman -S i3-wm xorg-xinit" - СКАЧАЕТ ОКОННЫЙ МЕНЕДЖЕР И XINIT ДЛЯ ЗАПУСКА ЧЕРЕЗ КОНСОЛЬ;
	3) "sudo cp /etc/X11/xinit/xinitrc ~/.xinitrc" - КОПИРУЕМ КОНФИГ В ДОМАШНИЙ КАТАЛОГ;
	4) "sudo nano ~/.xinitrc" >> "exec i3" - ОТКРОЕМ И ДОБАВИМ В НАЧАЛО ЗАПУСК I3;
	5) "sudo pacman -S alacritty" - СКАЧАЕТ ТЕРМИНАЛ ДЛЯ ДАЛЬНЕЙШЕЙ НАСТРОЙКИ ВНУТРИ I3.

СПИСОК ПО И УСТАНОВКА:

1. "sudo pacman -Syu" - ОБНОВЛЕНИЕ ВСЕХ ПАКЕТОВ, ОШИБКА КОТОРАЯ МОЖЕТ ВОЗНИКНУТЬ:
	1) "sudo pacman-key --edit-key "key" save" - В СЛУЧАЕ КОГДА НЕ ХОЧЕТ ОБНОВЛЯТЬ ИЗ-ЗА НЕПРАВИЛЬНОЙ ПОДПИСИ, ВВЕСТИ KEY(НАЙТИ В ОШИБКЕ "sudo pacman -Syu");
	2) "sudo pacman-key --init && sudo pacman-key --populate archlinux && sudo pacman-key --refresh-keys" - ЗАПУСК ПРОВЕРКИ И ОБНОВЛЕНИЯ КЛЮЧЕЙ GPG(ПРОВЕРЯЮТ ПОДПИСИ У ПАКЕТОВ).

2. YAY - ПОЗВОЛЯЕТ УСТАНАВЛИВАТЬ ПАКЕТЫ ИЗ AUR
	1) "sudo pacman -S git base-devel" - ИНСТРУМЕНТЫ ДЛЯ СКАЧИВАНИЯ И РАБОТЫ YAY;
	2) КЛОНИРУЕМ YAY ИЗ РЕПОЗИТОРИЯ AUR С ПОМОЩЬЮ ПРЯМОЙ 	ССЫЛКИ: "git clone https://aur.archlinux.org/yay.git";
	3) "cd yay" И "makepkg -si" - ЗАЙДЕТ В ПАПКУ,РАСПАКУЕТ И УСТАНОВИТ YAY;
	4) "~/.cache/yay:$" - РАСПОЛОЖЕНИЕ СОБРАННЫХ ПАКЕТОВ;
	5) КОМАНДЫ ДЛЯ YAY:
	"yay -Syu" ОБНОВИТ ВСЕ ПАКЕТЫ В СИСТЕМЕ;
	"yay -Yc" ОЧИСТИТ НЕЖЕЛАТЕЛЬНЫЕ ЗАВИСИМОСТИ;
	"yay -Ps" ВЫВОДИТ СИСТЕМНУЮ СТАТИСТИКУ.

3. УСТАНОВКА ДРАЙВЕРОВ:
	1) "lspci -v | grep -A1 -e VGA -e 3D" ОПРЕДЕЛИТ ВИДЕОКАРТУ;
	2) "sudo pacman -S mesa" - ОСНОВА;
	3) "sudo pacman -S lib32-mesa" - ПОДДЕРЖКА 32-БИТНЫХ ПРИЛОЖЕНИЙ;
	4) "sudo pacman -S xf86-video-amdgpu" - ОБЕСПЕЧИВАЕТ 2D УСКОРЕНИЕ, ОСНОВНОЙ ДРАЙВЕР.

4. УСТАНОВКА I3-WM(ОКОННЫЙ МЕНЕДЖЕР):
	1) "sudo pacman -S xorg-server xorg-drivers" - УСТАНАВЛИВАЕМ ИКСЫ И СВОБОДНЫЕ ДРАЙВЕРА(ШТУКА КОТОРАЯ РЕАЛИЗУЕТ ЭКРАН);
	2) "sudo pacman -S i3-wm xorg-xinit" - СКАЧАЕТ ОКОННЫЙ МЕНЕДЖЕР И XINIT ДЛЯ ЗАПУСКА ЧЕРЕЗ КОНСОЛЬ(XINIT ЗАПУСКАЕТ В ИКСАХ ОКОННЫЙ МЕНЕДЖЕР);
	3) "sudo cp /etc/X11/xinit/xinitrc ~/.xinitrc" - КОПИРУЕМ КОНФИГ В ДОМАШНИЙ КАТАЛОГ;
	4) "sudo nano ~/.xinitrc" >> "exec i3" - ОТКРОЕМ И ДОБАВИМ В НАЧАЛО ЗАПУСК I3;
	5) "sudo pacman -S alacritty" - СКАЧАЕТ ТЕРМИНАЛ ДЛЯ ДАЛЬНЕЙШЕЙ НАСТРОЙКИ ВНУТРИ I3.

5. НАЧАЛО РАБОТЫ В I3:
	1) НАЖМЕМ "Enter" - СОЗДАНИЕ КОНФИГА ПО УМОЛЧАНИЮ;
	2) ВЫБЕРЕМ КНОПКУ MOD - ВЛИЯЕТ НА ТО КАК БУДЕМ ИСПОЛЬЗОВАТЬ СОЧЕТАНИЯ КЛАВИШ(СМОТРИ ДАЛЬШЕ ИЛИ "КОМАНДЫ DMENU";
	3) ДЛЯ БОЛЕЕ БЫСТРОЙ УСТАНОВКИ:
	WIN+ENTER - ОТКРЫТЬ ТЕРМИНАЛ(2 ХВАТИТ)
	WIN+S - ТОЛЬКО ОДНО ОКНО НА ЭКРАНЕ
	WIN+E - ВЕРНУТЬ ТО ЧТО БЫЛО(ЕСЛИ УДОБНО).

6. УСТАНОВИМ ПО, КОТОРОЕ БУДЕМ ИСПОЛЬЗОВАТЬ В "НАСТРОЙКЕ I3":
 	1) "yay -S flameshot" - ПРОГРАММА ДЛЯ СКРИНШОТОВ;
	2) "sudo pacman -S pavucontrol pulseaudio" - ПЕРВЫЙ ПАКЕТ ЭТО ГРАФИЧЕСКАЯ УТИЛИТА ДЛЯ УПРАВЛЕНИЯ ЗВУКОМ, А ВТОРАЯ ЭТО СЕРВЕР НЕОБХОДИМЫЙ ДЛЯ ЗАПУСКА ДАННОГО ПАКЕТА;
	3) "yay -S thorium-browser-bin" - УСТАНОВКА БРАУЗЕРА THORIUM;
	4) "sudo pacman -S dmenu" - УСТАНОВКА/ОБНОВЛЕНИЕ МЕНЕДЖЕРА ОКОН(СМОТРИ "КОМАНДЫ DMENU", МОЖНО УДОБНО ВЫЗЫВАТЬ ЛЮБЫЕ ПРОГРАММЫ);
	5) "sudo pacman -S pcmanfm" - УСТАНОВКА ФАЙЛОВОГО МЕНЕДЖЕРА:
		1) "ПРАВКА" -> "ПАРАМЕТРЫ" -> "РАСШИРЕННЫЕ" И УКАЗАТЬ "alacritty" КАК ЭМУЛЯТОР ТЕРМИНАЛА.
	6) "yay -S VLC" - УНИВЕРСАЛЬНЫЙ ПРОИГРЫВАТЕЛЬ, ДОЛГО		УСТАНАВЛИВАЕТСЯ, НЕТ РУССКОГО ЯЗЫКА;
	7) "yay -S code" - VISUAL STUDIO CODE;
	8) "sudo pacman -S galculator" - КАЛЬКУЛЯТОР;
	9) "sudo pacman -S unzip" - РАЗАРХИВИРУЕТ АРХИВЫ ТИПА ZIP;
	10) "sudo pacman -Syu" - ОБНОВЛЕНИЕ ВСЕХ ПАКЕТОВ(ПРИЛОЖЕНИЯ МОГУТ НЕ ЗАПУСТИТЬСЯ ИЗ-ЗА НЕОБНОВЛЕННЫХ ПАКЕТОВ).
